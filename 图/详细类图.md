```mermaid
classDiagram
    direction TB

    %% ---------------------------------------------------------
    %% Controller 层 (逻辑层)
    %% ---------------------------------------------------------
    class AuthController {
        +Login()
        +GetCurrentUser()
    }
    class StudentController {
        +GetStudents()
        +GetStudent()
        +CreateStudent()
        +UpdateStudent()
        +DeleteStudent()
    }
    class CourseController {
        +GetCourses()
        +GetCourse()
        +CreateCourse()
        +UpdateCourse()
        +DeleteCourse()
    }
    class GradeController {
        +GetGrades()
        +CreateGrade()
        +UpdateGrade()
        +DeleteGrade()
    }

    %% ---------------------------------------------------------
    %% Entity 层 (数据层)
    %% ---------------------------------------------------------
    class User {
        +uint ID
        +string Username
        +string Password
        +uint RoleID
        +bool IsActive
        +uint UserID
        +string UserType
    }
    class Student {
        +uint ID
        +string Name
        +string StudentID
        +string Gender
        +int Age
        +string Email
        +string Phone
        +string Address
    }
    class Course {
        +uint ID
        +string CourseName
        +uint TeacherID
        +float64 Credits
    }
    class Grade {
        +uint ID
        +uint StudentID
        +uint CourseID
        +string ScoreType
        +float64 Score
    }

    %% ---------------------------------------------------------
    %% 关系定义
    %% ---------------------------------------------------------
    AuthController ..> User : uses
    StudentController ..> Student : uses
    CourseController ..> Course : uses
    GradeController ..> Grade : uses
    GradeController ..> Student : check
    GradeController ..> Course : check

    User "1" --> "1" Student : 关联
    Student "1" --> "0..*" Grade : 拥有
    Course "1" --> "0..*" Grade : 包含
```
